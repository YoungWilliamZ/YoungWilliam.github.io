## 代码

实现的有点粗暴，没有考虑一些边界值，但题目给了边界值，所以不用写也能过。

``` cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        if (head->next == NULL && n == 1) {
            return NULL;
        }
        
        ListNode* rm = head;
        ListNode* curr = head;
        ListNode* last;
        for (int i = 0;curr != NULL && i < n; i++) {
            curr = curr->next;
        }
        while (curr != NULL) {
            curr = curr->next;
            last = rm;
            rm = rm->next;
        }
        if (n == 1 && last != NULL) {
            last->next = NULL;
        } else {
            rm->val = rm->next->val;
            rm->next = rm->next->next;
        }
        
        return head;
    }
};
```
